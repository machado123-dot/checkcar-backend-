const express = require("express");
const puppeteer = require("puppeteer");
const cors = require("cors");

const app = express();
app.use(cors());

function estado(data) {
  const hoje = new Date();
  const validade = new Date(data);

  return {
    data,
    valido: validade > hoje
  };
}

async function consultarIPO(matricula) {
  const browser = await puppeteer.launch({
    headless: true,
    args: ["--no-sandbox", "--disable-setuid-sandbox"]
  });

  const page = await browser.newPage();

  try {
    await page.goto("https://eportugal.gov.pt/servicos/consultar-inspecao-automovel", {
      waitUntil: "networkidle2"
    });

    await page.type('input[name="matricula"]', matricula);
    await page.click("button");

    await page.waitForTimeout(3000);

    const data = await page.evaluate(() => {
      const el = document.querySelector(".validade");
      return el ? el.innerText : null;
    });

    await browser.close();

    if (!data) throw new Error("Sem dados");

    return estado(data);

  } catch (err) {
    await browser.close();
    return { erro: "Falha ao obter IPO" };
  }
}

app.get("/api/:matricula", async (req, res) => {
  const ipo = await consultarIPO(req.params.matricula);

  res.json({
    matricula: req.params.matricula,
    ipo
  });
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log("API running"));
